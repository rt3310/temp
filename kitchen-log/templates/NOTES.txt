# templates/NOTES.txt
1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.spring.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "kitchen-log.fullname" . }}-spring)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Access your application at: http://$NODE_IP:$NODE_PORT"
  
  Or use any of your cluster nodes:
  echo "http://<any-node-ip>:$NODE_PORT"
  
{{- else if contains "LoadBalancer" .Values.service.spring.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "kitchen-log.fullname" . }}-spring'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "kitchen-log.fullname" . }}-spring --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.spring.port }}
{{- else if contains "ClusterIP" .Values.service.spring.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "kitchen-log.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=spring" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. MySQL connection:
  MySQL is running as a separate StatefulSet with ClusterIP service.
  
  To connect to MySQL from within the cluster:
  Host: {{ include "kitchen-log.fullname" . }}-mysql-0.{{ include "kitchen-log.fullname" . }}-mysql.{{ .Release.Namespace }}.svc.cluster.local
  Port: 3306
  Database: {{ .Values.mysql.database }}
  User: {{ .Values.mysql.user }}
  
  To connect to MySQL from your local machine:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "kitchen-log.fullname" . }}-mysql 3306:3306
  
  Then connect using:
  Host: localhost
  Port: 3306
  Database: {{ .Values.mysql.database }}
  User: {{ .Values.mysql.user }}

3. Check the status of your deployment:
  kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}
  
  kubectl get svc -n {{ .Release.Namespace }}
  
4. View logs:
  Spring application logs:
    kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/component=spring -f
  
  MySQL logs:
    kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/component=mysql -f

5. Test the Spring API:
{{- if contains "NodePort" .Values.service.spring.type }}
  # Health check
  curl http://<node-ip>:{{ .Values.service.spring.nodePort }}/actuator/health
  
  # Get node IPs
  kubectl get nodes -o wide
{{- end }}
